create database if not exists redis_monitor;
use redis_monitor;
create table if not exists node_info(
    id int auto_increment primary key,
    connected_clients int,
    blocked_clients int,
    used_memory int,
    used_memory_rss int,
    used_memory_peak int,
    mem_fragmentation_ratio float,
    aof_enabled int,
    total_connections_received int,
    total_commands_processed int,
    instantaneous_ops_per_sec int,
    total_net_input_bytes int,
    total_net_output_bytes int,
    instantaneous_input_kbps float,
    instantaneous_output_kbps float,
    rejected_connections int,
    sync_full int,
    sync_partial_ok int,
    sync_partial_err int,
    expired_keys int,
    evicted_keys int,
    keyspace_hits int,
    keyspace_misses int,
    pubsub_channels int,
    pubsub_patterns int,
    latest_fork_usec int,
    migrate_cached_sockets int,
    used_cpu_sys float,
    used_cpu_user float,
    used_cpu_sys_children float,
    used_cpu_user_children float,
    clusterid varchar(50),
    nodeid varchar(255),
    host varchar(30),
    ip varchar(25),
    port int,
    day int,
    hour int,
    minute int,
    add_time int
);
